CREATE TABLE movies (
mid INT UNIQUE NOT NULL,
title VARCHAR(250),
year DATE,
rating REAL,
num_ratings INT,
PRIMARY KEY (mid)
);

CREATE TABLE actors (
mid INT,
name VARCHAR(50),
cast_position INT,
FOREIGN KEY (mid) REFERENCES movies(mid),
PRIMARY KEY (mid, name)
);

CREATE TABLE genres (
mid INT,
genre VARCHAR(50),
FOREIGN KEY (mid) REFERENCES movies(mid),
PRIMARY KEY (mid, genre)
);

CREATE TABLE tag_names (
tid INT UNIQUE NOT NULL,
tag VARCHAR(50),
PRIMARY KEY (tid)
);


CREATE TABLE tags (
mid INT,
tid INT,
FOREIGN KEY (mid) REFERENCES movies(mid),
FOREIGN KEY (tid) REFERENCES tag_names(tid),
PRIMARY KEY (mid, tid)
);
