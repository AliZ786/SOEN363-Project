create table Cities
(
    cid INTEGER,
    city CHAR(50),
    country CHAR(50),
    capital CHAR(8),
    population INTEGER,
    stationID CHAR(20),
    PRIMARY KEY (cid)
);

create table WeatherConditions
(
    wid INTEGER,
    stationID CHAR(20),
    date DATE,
    precipitation INTEGER,
    snowfall INTEGER,
    snow_depth INTEGER,
      PRIMARY KEY (wid)
);

create table Temperature
(
    tid INTEGER,
    stationID CHAR(20),
    date DATE,
    average_temp INTEGER,
    max_temp INTEGER,
    min_temp INTEGER,
    PRIMARY KEY (tid)
);
